class Solution {
    public int lengthOfLongestSubstring(String s) {
        if (s.length()==0) return 0;
        HashMap<Character, Integer> map = new HashMap<Character, Integer>();
        int max = 0;
        int left = 0;
        for(int i = 0; i < s.length(); i ++){
            if(map.containsKey(s.charAt(i))){
                left = Math.max(left,map.get(s.charAt(i)) + 1);
            }
            map.put(s.charAt(i),i);
            max = Math.max(max,i-left+1);
        }
        return max;
    }
}

    for(int i = 0; i < numRows; i++) rows.add(new StringBuilder());
    int i = 0, flag = -1;

中序遍历：每次pop之后，都需要去判断右边是否为空，然后再次向下push

class Solution {
    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> result=new ArrayList<>();
        if (root==null){
            return result;
        }
        Stack<TreeNode> myStack=new Stack<>();
        TreeNode cur=root;
        while (cur!=null || !myStack.isEmpty())
        {
            while (cur!=null)
            {
                myStack.push(cur);
                cur=cur.left;
            }
            TreeNode node=myStack.pop();
            result.add(node.val);
            if (node.right!=null){
                cur=node.right;
            }
        }
        return result;
    }

class Solution {
    public List<Integer> postorderTraversal(TreeNode root) {
        List<Integer> result=new ArrayList<>();
        if (root == null) {
            return result;
        }
        Stack<TreeNode> myStack=new Stack<TreeNode>();
        TreeNode cur=root;
        TreeNode pre=null;
        while (!myStack.empty()  || cur!=null){
            while (cur!=null){
                myStack.push(cur);
                cur=cur.left;
            }
            TreeNode node=myStack.pop();
            if (node.right!=null && node.right!= pre)
            {
                myStack.push(node);
                cur=node.right;
            }else {
                result.add(node.val);
                pre=node;
            }
        }
        return result;
    }
}

层次遍历：就是通过队列，一层一层的加入：

class Solution {
    public List<List<Integer>> levelOrder(TreeNode root) {
        List<List<Integer>> result=new ArrayList<>();
        if (root==null){
            return result;
        }
        Queue<TreeNode> treeNodeQueue=new LinkedList<TreeNode>();
        treeNodeQueue.offer(root);
        while (!treeNodeQueue.isEmpty()){
            List<Integer> integerList=new ArrayList<>();
            int queueSize=treeNodeQueue.size();
            for (int i = 0; i < queueSize; i++) {
                TreeNode pollNode = treeNodeQueue.poll();
                integerList.add(pollNode.val);
                if (pollNode.left!=null){
                    treeNodeQueue.offer(pollNode.left);
                }
                if (pollNode.right!=null){
                    treeNodeQueue.offer(pollNode.right);
                }
            }
            result.add(integerList);
        }
        return result;
    }
}
