ListNode successor = null; // 后驱节点

// 反转以 head 为起点的 n 个节点，返回新的头结点
ListNode reverseN(ListNode head, int n) {
    if (n == 1) {
        // 记录第 n + 1 个节点
        successor = head.next;
        return head;
    }
    // 以 head.next 为起点，需要反转前 n - 1 个节点
    ListNode last = reverseN(head.next, n - 1);

    head.next.next = head;
    // 让反转之后的 head 节点和后面的节点连起来
    head.next = successor;
    return last;
}

反转区间：

ListNode reverseBetween(ListNode head, int m, int n) {
    // base case
    if (m == 1) {
        return reverseN(head, n);
    }
    // 前进到反转的起点触发 base case
    head.next = reverseBetween(head.next, m - 1, n - 1);
    return head;
}

/** 反转区间 [a, b) 的元素，注意是左闭右开 */
ListNode reverse(ListNode a, ListNode b) {
    ListNode pre, cur, nxt;
    pre = null; cur = a; nxt = a;
    // while 终止的条件改一下就行了
    while (cur != b) {
        nxt = cur.next;
        cur.next = pre;
        pre = cur;
        cur = nxt;
    }
    // 返回反转后的头结点
    return pre;
}

主逻辑：

ListNode reverseKGroup(ListNode head, int k) {
    if (head == null) return null;
    // 区间 [a, b) 包含 k 个待反转元素
    ListNode a, b;
    a = b = head;
    for (int i = 0; i < k; i++) {
        // 不足 k 个，不需要反转，base case
        if (b == null) return head;
        b = b.next;
    }
    // 反转前 k 个元素
    ListNode newHead = reverse(a, b);
    // 递归反转后续链表并连接起来
    a.next = reverseKGroup(b, k);
    return newHead;
}

10 正则表达式 动态规划 hard

​ 对于这种匹配的类型，就应该想到用动态规划，动态规划最重要的就是状态的转换，所以这道题也需要分清状态的变化，其实就是各种情况的讨论。

​ 因为有只有一个字符的原因，所以这里就从i,j=1开始遍历，结束为i<=s.length。所以s[i-1]表示第i个字符，所以这里boolean [][] dp = new boolean[n+1] [m+1];

中间状态转移：

​ 遍历s和p的字符，dp[i] [j]表示s串前i个字符能否被p串前 j 个字符匹配

​ 首先判断，s[i-1]==p[j-1],

​ 如果相等，说明是普通字符相等，就是dp[i] [j]=dp[i-1] [j-1]；

​ 如果不相等，

​ 如果p[j-1]是‘.’,那么也是dp[i] [j]=dp[i-1] [j-1]；

​ 如果p[j-1]是‘*’，则又要分几种情况，需要判断 *前面的字符和s的字符是否匹配 即p[j-2]==s[i-1];

​ 如果不匹配，即s[i-1] != p[j-2] && p[i-2] != ‘.’，则只能 *将前面那个字符匹配0次，所以 dp[i] [j]=dp[i] [j-2]；

​ 如果匹配，即s[i-1] == p[j-2] || p[i-2] == ‘.’又要分几种情况，

​ 有可能s中只有一个字符例如：s：abcd 。p：abcd*，此时 dp[i] [j] = dp[i-1] [j-2] 有可能s中只有多个字符例如：s：abcdddd 。p：abcd ，此时dp[i] [j] = dp[i-1] [j] 有可能s中只有0个字符例如：s：abcd 。p：abcdd，此时 dp[i] [j] = dp[i] [j-2]

        int p = 0, q = 0, r = 1;
        for (int i = 1; i <= n; ++i) {
            p = q;

public class main3 {
    public static void main(String[] args) {
        int[] array = {6,172,113,11,23};
        quickSort(array, 0, array.length -1);
        System.out.println("排序后的结果");
        System.out.println(Arrays.toString(array));
    }

    private static int quickSortFun(int[] array, int low, int high) {
        int pivot=array[high];
        int pointer=low;
        for (int i = low; i < high; i++) {
            if (array[i]<=pivot){
                int tmp=array[i];
                array[i]=array[pointer];
                array[pointer]=tmp;
                pointer++;
            }
        }
        int tmp=array[high];
        array[high]=array[pointer];
        array[pointer]=tmp;
        return pointer;
    }

    public static void quickSort(int[] array, int low, int high){
        if (low<high){
            int mid = quickSortFun(array, low, high);
            quickSortFun(array,low,mid-1);
            quickSortFun(array,mid+1,high);
        }
    }
}
